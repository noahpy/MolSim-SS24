23inline std::vector<Position> coordinateToPosition(
24 std::array<size_t, 3> coordinate,
25 std::array<size_t, 3> gridDimensions,
26 bool keep3D = true,
27 bool isHalo = false)
28{
29 size_t offset = isHalo ? 0 : 1;
30 std::vector<Position> positions;
31 if (coordinate[0] == offset) {
32 positions.push_back(Position::LEFT);
33 }
34 if (coordinate[0] == gridDimensions[0] - 1 - offset) {
35 positions.push_back(Position::RIGHT);
36 }
37 if (coordinate[1] == gridDimensions[1] - 1 - offset) {
38 positions.push_back(Position::TOP);
39 }
40 if (coordinate[1] == offset) {
41 positions.push_back(Position::BOTTOM);
42 }
43 if (coordinate[2] == gridDimensions[2] - 1 - offset && keep3D) {
44 positions.push_back(Position::FRONT);
45 }
46 if (coordinate[2] == offset && keep3D) {
47 positions.push_back(Position::BACK);
48 }
49
50 if (positions.empty())
51 throw std::invalid_argument("Coordinate is not on the boundary");
52
53 return positions;
54}

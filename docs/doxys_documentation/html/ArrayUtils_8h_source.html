97template <class Container>
98[[nodiscard]] std::string to_string(
99 const Container& container,
100 const std::string& delimiter = ", ",
101 const std::array<std::string, 2>& surround = { "[", "]" })
102{
103 auto iter = std::cbegin(container);
104 const auto end = std::cend(container);
105 if (iter == end) {
106 return surround[0] + surround[1];
107 }
108 std::ostringstream strStream;
109 strStream << surround[0] << *iter;
110 for (++iter; iter != end; ++iter) {
111 strStream << delimiter << *iter;
112 }
113 strStream << surround[1];
114 return strStream.str();
115}

130template <class Container, class F>
131inline Container elementWisePairOp(const Container& lhs, const Container& rhs, F binaryFunction)
132{
133 Container ret = lhs;
134 auto retIter = std::begin(ret);
135 auto lhsIter = std::cbegin(lhs);
136 const auto lhsEnd = std::cend(lhs);
137 auto rhsIter = std::cbegin(rhs);
138 const auto rhsEnd = std::cend(rhs);
139
140 for (; lhsIter != lhsEnd and rhsIter != rhsEnd; ++lhsIter, ++rhsIter, ++retIter) {
141 *retIter = binaryFunction(*lhsIter, *rhsIter);
142 }
143
144 return ret;
145}

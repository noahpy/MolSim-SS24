91template <class Container>
92[[nodiscard]] std::string to_string(
93 const Container& container,
94 const std::string& delimiter = ", ",
95 const std::array<std::string, 2>& surround = { "[", "]" })
96{
97 auto iter = std::cbegin(container);
98 const auto end = std::cend(container);
99 if (iter == end) {
100 return surround[0] + surround[1];
101 }
102 std::ostringstream strStream;
103 strStream << surround[0] << *iter;
104 for (++iter; iter != end; ++iter) {
105 strStream << delimiter << *iter;
106 }
107 strStream << surround[1];
108 return strStream.str();
109}

124template <class Container, class F>
125inline Container elementWisePairOp(const Container& lhs, const Container& rhs, F binaryFunction)
126{
127 Container ret = lhs;
128 auto retIter = std::begin(ret);
129 auto lhsIter = std::cbegin(lhs);
130 const auto lhsEnd = std::cend(lhs);
131 auto rhsIter = std::cbegin(rhs);
132 const auto rhsEnd = std::cend(rhs);
133
134 for (; lhsIter != lhsEnd and rhsIter != rhsEnd; ++lhsIter, ++rhsIter, ++retIter) {
135 *retIter = binaryFunction(*lhsIter, *rhsIter);
136 }
137
138 return ret;
139}

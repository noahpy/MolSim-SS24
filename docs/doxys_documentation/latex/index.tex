\texorpdfstring{$\ast$}{*}\texorpdfstring{$\ast$}{*}{\ttfamily \doxylink{classPhysicsStrategy}{Physics\+Strategy}}\texorpdfstring{$\ast$}{*}\texorpdfstring{$\ast$}{*}


\begin{DoxyItemize}
\item Strategy interface allows for interchangable / compareable implementation of the physics
\item Realistically, velocities and positions are calculated according to Störmer-\/\+Verlet and are not changed (only parallelized later on)
\item Interchangeable are the force calculations -\/\texorpdfstring{$>$}{>} Essentially each simulation has its own implementation of the forces
\end{DoxyItemize}

\texorpdfstring{$\ast$}{*}\texorpdfstring{$\ast$}{*}{\ttfamily \doxylink{classCellGrid}{Cell\+Grid}}\texorpdfstring{$\ast$}{*}\texorpdfstring{$\ast$}{*}


\begin{DoxyItemize}
\item The MD-\/\+Simulation uses the linked-\/cell algorithm starting from the \doxylink{classLinkedLennardJonesSimulation}{Linked\+Lennard\+Jones\+Simulation}
\item The algorithm scales in {\bfseries{O(n)}} relative to the number of particles
\item Assumes a certain {\itshape cutoff} distance after which the forces are neglected
\end{DoxyItemize}

\texorpdfstring{$\ast$}{*}\texorpdfstring{$\ast$}{*}{\ttfamily \doxylink{classThermostat}{Thermostat}}\texorpdfstring{$\ast$}{*}\texorpdfstring{$\ast$}{*}


\begin{DoxyItemize}
\item Used to cool / heat / maintain the temperature of the simulation
\item There are two thermostats available
\item The classical thermostat will reduce the temperature without regarding the mean velocity of the system, while the individual thermostat does
\end{DoxyItemize}
